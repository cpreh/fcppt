function(
	make_fcppt_example
	SUBNAME
	MAINNAME
	ADDITIONAL_DEPS
	ADDITIONAL_DEFINES
)
	if(
		"${SUBNAME}"
		STREQUAL
		""
	)
		set(
			FCPPT_EXAMPLE_NAME
			example_${MAINNAME}
		)
	else()
		set(
			FCPPT_EXAMPLE_NAME
			example_${SUBNAME}_${MAINNAME}
		)
	endif()

	add_executable(
		${FCPPT_EXAMPLE_NAME}
		${MAINNAME}.cpp
	)

	set_target_properties(
		${FCPPT_EXAMPLE_NAME}
		PROPERTIES
		COMPILE_FLAGS
		"${ADDITIONAL_DEFINES}"
	)

	fcppt_link_target(
		fcppt_core
	)

	target_link_libraries(
		${FCPPT_EXAMPLE_NAME}
		${fcppt_core_TARGET}
		${FCPPT_DEFAULT_LINK_TARGET}
		${ADDITIONAL_DEPS}
	)
endfunction()

function(
	add_fcppt_example
	MAINNAME
	ADDITIONAL_DEPS
	ADDITIONAL_DEFINES
)
	make_fcppt_example(
		""
		${MAINNAME}
		"${ADDITIONAL_DEPS}"
		"${ADDITIONAL_DEFINES}"
	)
endfunction()
